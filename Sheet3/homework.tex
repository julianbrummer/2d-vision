%
% This environment takes an optional argument. When given, it will adjust the
% Task counter. This is useful for when the Tasks given for your
% assignment aren't sequential. See the last 3 Tasks of this template for an
% example.
%
\newenvironment{homeworkTask}[1][-1]{
  \ifnum#1>0
    \setcounter{homeworkTaskCounter}{#1}
  \fi
  \section{Task \arabic{homeworkTaskCounter}}
  \setcounter{partCounter}{1}
  \enterTaskHeader{homeworkTaskCounter}
}{
  \exitTaskHeader{homeworkTaskCounter}
}
